3    import javafx.scene.canvas.GraphicsContext;
4    import javafx.scene.paint.Color;
5    
6    public class MyRectangle extends MyShape {
7    
8        private double w1, h1;
9    
10       // constructor
11       public MyRectangle(double posX, double posY, double w1, double h1, Color C){ // posX is upper left of x, PosY is upper left of y. w1 is width of rectangle, h1 is height.
12           this.setX(posX);
13           this.setY(posY);
14           this.setWidth(w1);
15           this.setHeight(h1);
16           this.setColor(C);
17   
18       }
19   
20       public void setWidth(double X){ this.w1 = X; }
21   
22       public double getWidth(){ return w1; }
23   
24       public void setHeight(double X){ this.h1 = X; }
25   
26       public double getHeight(){ return h1; }
27   
28       @Override
29       public String toString(){
30           return "Width: " + getWidth() + "\n" +"Height: " + getHeight() + "\n" + "Area: " + getWidth()*getHeight() + "\n" + "Perimeter: " + 2*(getWidth()+getHeight());
31       }
32   
33       public void draw(GraphicsContext gc){
34           gc.setFill(getColor());
35           // This is so the rectangle is in the center of the canvas.
36           double w = getWidth()/2;
37           double h = getHeight()/2;
38           gc.fillRect(getX()-w,getY()-h,getWidth(),getHeight());
39       }
40   }
