15       public void start(Stage primaryStage)
16       {
17           Canvas canvas = new Canvas(800,400);
18           GraphicsContext gc = canvas.getGraphicsContext2D();
19           FlowPane flow = new FlowPane();
20           Scene scene = new Scene(flow, canvas.getWidth(), canvas.getHeight());
21   
22           // The Colors I created for the shapes using myColor
23           myColor C = myColor.Blue;
24           myColor G = myColor.Green;
25           myColor S = myColor.Silver;
26           myColor B = myColor.Black;
27   
28           //This is how to fit the inner rectangles inside the area of the oval. uses the formula x = a/sqrt(2) and y = b/sqrt(2).
29           double a = canvas.getWidth()/2;
30           double b = canvas.getHeight()/2;
31   
32           //creates shapes
33           MyRectangle rec1 = new MyRectangle(canvas.getWidth()/2,canvas.getHeight()/2, canvas.getWidth()/2,canvas.getHeight()/2, C.returnColour());
34           MyLine line1 = new MyLine(0,0,canvas.getWidth(),canvas.getHeight(),B.returnColour());
35           MyOval oval1 = new MyOval(canvas.getWidth()/2,canvas.getHeight()/2, a,b, S.returnColour());
36           MyShape rec2 = new MyRectangle(canvas.getWidth()/2,canvas.getHeight()/2, a = (a)/sqrt(2),b = (b)/sqrt(2), G.returnColour());
37           MyShape oval2 = new MyOval(canvas.getWidth()/2,canvas.getHeight()/2, a,b, C.returnColour());
38           MyShape rec3 = new MyRectangle(canvas.getWidth()/2,canvas.getHeight()/2, a = (a)/sqrt(2),b = (b)/sqrt(2), S.returnColour());
39           MyShape oval3 = new MyOval(canvas.getWidth()/2,canvas.getHeight()/2, a,b, G.returnColour());
40   
41           //Draws shape on canvas
42           rec1.draw(gc);
43           oval1.draw(gc);
44           rec2.draw(gc);
45           oval2.draw(gc);
46           rec3.draw(gc);
47           oval3.draw(gc);
48           line1.draw(gc);
49   
50           flow.getChildren().addAll(canvas);
51           primaryStage.setTitle("Assignment 1");
52           primaryStage.setScene(scene);
53           primaryStage.show();
54   
55       }
