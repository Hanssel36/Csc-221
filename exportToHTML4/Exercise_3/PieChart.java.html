10   public class PieChart {
11   private double w1,h1;
12   private LinkedHashMap<Character,Double> prop; // holds the probability of the letters
13   
14           public PieChart( double w1, double h1, LinkedHashMap<Character,Double> prop)
15           {
16               setHeight(h1);
17               setWidth(w1);
18               setAngle(prop);
19           }
20   
21       public void setWidth(double X){
22           this.w1 = X;
23       }
24   
25       public double getWidth(){
26           return w1;
27       }
28   
29       public void setHeight(double X){
30           this.h1 = X;
31       }
32   
33       public double getHeight(){
34           return h1;
35       }
36   
37       public void setAngle(LinkedHashMap<Character,Double> prop){
38               this.prop = prop;
39       }
40   
41       public double[] getAngle(){
42           double[] angle; // array of angles
43           angle = new double[prop.size()];
44           int count =0;
45           for(double vals: prop.values()){
46               angle[count] = 360*vals;
47               count =count + 1;
48           }
49               return angle;
50       }
51   
52       // For Project this was not needed
53       public double RestOfProp(){
54   
55   
56               double sum = 0.0;
57               for (double vals: prop.values()){
58                   sum += vals;
59               }
60               return 1.0-sum;
61       }
62   
63   public void draw(GraphicsContext gc, Map<Character,Double> gradeNum){
64       Random rand = new Random();
65   
66   //Color c = new Color(r,g,b);
67       double s = 0; //used for helping the pie segments find their right position
68   
69       double pad = 20; // used for padding between the labels and their colors.
70       for(int i = 0; i < getAngle().length; i++) {
71           int r = rand.nextInt(256); // goes from 0 to 255
72           int g = rand.nextInt(256);
73           int b = rand.nextInt(256);
74   
75           gc.setFill(Color.rgb(r,g,b));
76           gc.fillArc(75 , 150 , getWidth(), getHeight(), s, getAngle()[i], ArcType.ROUND);
77           Double value = (Double) gradeNum.values().toArray()[i]; // this is to turn Double
78           Integer va = value.intValue();// turns into an Integer
79           gc.fillText(prop.keySet().toArray()[i] + ", "+ va,getWidth()+75, pad);
80   
81           s = s +getAngle()[i];
82           pad +=20;
83       }
84   
85       gc.strokeArc(75,150,getWidth(),getHeight(),0,360, ArcType.OPEN); // This creates the circle of the piechart
86   }
87   }
