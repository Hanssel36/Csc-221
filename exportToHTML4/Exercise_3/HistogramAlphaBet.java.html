11   public class HistogramAlphaBet {
12       private int n; // used to determine n probabilty of characters.
13       public Map<Character,Double> mysort; // used temporily to sort
14       private LinkedHashMap<Character,Double> myprop = new LinkedHashMap<>(); // returns the probability as the values and the letters as the keys.
15   
16       public HistogramAlphaBet(int n, Map<Character, Double> temphash)
17       {
18           this.n = n;
19   
20   
21           Map<Character, Double> mysort = temphash
22                   .entrySet()
23                   .stream()
24                   .collect(toMap(e -> e.getKey(), e -> e.getValue(), (e1, e2) -> e2, LinkedHashMap::new));
25   
26           this.mysort = mysort;
27   
28       }
29   
30       // This will calculate the probabilty
31       public LinkedHashMap<Character,Double> calcP(){
32           double sum = 0; // Hold sum of all letters
33   
34           // Calculates the the sum of values or the letter counts
35           for(double vals: mysort.values()){
36               sum = sum + vals;
37           }
38           int count = 0;
39   
40           Iterator<Map.Entry<Character, Double>> itr = mysort.entrySet().iterator();
41   
42           while(itr.hasNext())
43           {
44               if(count == n){
45                   break;
46               }
47   
48               Map.Entry<Character, Double> entry = itr.next();
49               myprop.put(entry.getKey(),entry.getValue()/sum);
50               count =count + 1;
51           }
52   
53           
54           return myprop;
55       }
