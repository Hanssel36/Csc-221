22           return "Length " + sqrt(pow((getX2()-getX()),2) + pow((getY2()-getY()),2)) + "\n" // Length uses distance formula sqrt((x2-x1)^2 + (y2-y1)^2)
23                   + "Angle " + toDegrees(atan((getY2()-getY())/(getX2()-getX())));
24       }

38           if(F.getX() == S.getX()) {
39               return true;

41           if(F.getY() == S.getY()) {
42               return true;
43           }
44           else {
45               return false;
46           }
